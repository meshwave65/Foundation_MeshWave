# .github/workflows/run_agent_app_core.yml
name: Run Agent AppCore

on:
  workflow_dispatch: # Disparo manual
  # schedule:
  #   - cron: '0 0 * * *' # Diariamente à meia-noite UTC

jobs:
  run_agent:
    runs-on: ubuntu-latest
    steps:
    - name: Checkout repository
      uses: actions/checkout@v3
      with:
        token: ${{ secrets.GITHUB_PAT }} # Ou usar GitHub App para autenticação

    - name: Setup Python
      uses: actions/setup-python@v4
      with:
        python-version: '3.9'

    - name: Install dependencies
      run: |
        pip install GitPython # e outras libs que o agente precise

    - name: Configure Git for agent
      run: |
        git config user.name "Agent AppCore Bot"
        git config user.email "agent-appcore@meshwave.com"

    - name: Run Agent AppCore
      env:
        # Passar segredos como variáveis de ambiente para o agente
        GITHUB_PAT: ${{ secrets.GITHUB_PAT }}
        # ... outras variáveis de ambiente que o agente precise
      run: |
        python agents/specialists/phase1/application/agent_app_core/src/main.py

    - name: Commit and Push changes
      run: |
        git add phases/phase1_foundation_integration/application/
        git commit -m "Agent AppCore: Automated update for App MeshWave module" || echo "No changes to commit"
        git push
