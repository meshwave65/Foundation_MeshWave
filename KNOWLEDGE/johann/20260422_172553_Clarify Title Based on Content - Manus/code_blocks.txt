<script setup>
import { ref, computed, onUnmounted } from 'vue';

const LITRES_PER_SECOND = 6.52;
const OLYMPIC_POOL_VOLUME = 2500000;

const accumulatedLitres = ref(0);
const simulationStarted = ref(false);
const simulationRunning = ref(false);
let intervalId = null;

const formattedLitres = computed(() => {
  return new Intl.NumberFormat('pt-BR').format(accumulatedLitres.value.toFixed(0));
});

const poolPercentage = computed(() => {
  if (accumulatedLitres.value === 0) return 0;
  return (accumulatedLitres.value / OLYMPIC_POOL_VOLUME) * 100;
});

const maskHeight = computed(() => {
  return 100 - poolPercentage.value;
});

const startSimulation = () => {
  if (intervalId) return;
  simulationStarted.value = true;
  simulationRunning.value = true;
  intervalId = setInterval(() => {
    accumulatedLitres.value += LITRES_PER_SECOND;
  }, 1000);
};

const stopSimulation = () => {
  clearInterval(intervalId);
  intervalId = null;
  simulationRunning.value = false;
};

const toggleSimulation = () => {
  if (!simulationStarted.value) {
    startSimulation();
  } else if (simulationRunning.value) {
    stopSimulation();
  } else {
    startSimulation();
  }
};

onUnmounted(() => {
  stopSimulation();
});
</script>
