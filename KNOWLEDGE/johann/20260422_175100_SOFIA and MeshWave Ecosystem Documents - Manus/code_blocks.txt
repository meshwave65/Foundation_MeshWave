import requests
import time
import sys
import os

sys.path.append(os.path.dirname(os.path.dirname(os.path.abspath(__file__))))
try:
    from architecture import SOFIA_API_BASE_URL
except ImportError as e:
    print(f"ERRO CRÍTICO: Falha na importação de um módulo essencial: {e}")
    sys.exit(1)

# --- Funções de API (sem alterações) ---
def find_task_by_status(status_id):
    try:
        response = requests.get(f"{SOFIA_API_BASE_URL}/tasks/")
        if response.status_code == 200:
            for task in response.json():
                if task.get('status', {}).get('status_id') == status_id:
                    return task
        return None
    except requests.exceptions.RequestException: return None

def update_task_status(task_id, new_status_id, persona_id=None):
    log_prefix = f"[RHAgent-{task_id}]"
    payload = {"status_id": new_status_id}
    if persona_id is not None: payload["persona_id"] = persona_id
    try:
        response = requests.patch(f"{SOFIA_API_BASE_URL}/tasks/{task_id}", json=payload)
        if response.status_code == 200:
            print(f"{log_prefix} ✅ Status atualizado para {new_status_id}.")
            return True
        print(f"{log_prefix} ❌ Falha ao atualizar status. Código: {response.status_code}, Resposta: {response.text}")
        return False
    except requests.exceptions.RequestException: return False

def get_enriched_description(task_id):
    try:
        response = requests.get(f"{SOFIA_API_BASE_URL}/tasks/{task_id}/blocks")
        if response.status_code == 200:
            for block in response.json():
                if block.get('block_type') == "EnrichedDescription":
                    return block.get('content')
        return None
    except requests.exceptions.RequestException: return None

def determine_persona(description):
    return 999999

import requests
import time
import os
import json
import random

try:
    from architecture import SOFIA_API_BASE_URL, GROQ_API_KEY, ARTIFACTS_DIR, GROQ_MODEL_POWERFUL, GROQ_MODEL_FAST
    from oracle_client import chamar_oraculo_json
except ImportError as e:
    print(f"ERRO CRÍTICO: Falha na importação de um módulo essencial: {e}")
    exit()

# --- Funções de API (sem alterações) ---
def find_and_claim_worker_task(session):
    try:
        response = session.get(f"{SOFIA_API_BASE_URL}/tasks/")
        if response.status_code == 200:
            for task in response.json():
                if task.get('status', {}).get('status_id') == 1:
                    task_id = task['task_id']
                    print(f"    [Worker-{task_id}] Tarefa com status 1 encontrada. Reivindicando (status 50)...")
                    claim_response = session.patch(f"{SOFIA_API_BASE_URL}/tasks/{task_id}", json={"status_id": 50})
                    if claim_response.status_code == 200:
                        task_details_response = session.get(f"{SOFIA_API_BASE_URL}/tasks/{task_id}")
                        if task_details_response.status_code == 200:
                            return task_details_response.json()
        return None
    except requests.exceptions.RequestException:
        return None

def update_task_status(session, task_id, new_status_id):
    try:
        print(f"    [API] Tentando atualizar task {task_id} para status {new_status_id}...")
        response = session.patch(f"{SOFIA_API_BASE_URL}/tasks/{task_id}", json={"status_id": new_status_id})
        response.raise_for_status()
        print(f"    [API] ✅ Sucesso.")
        return True
    except requests.exceptions.RequestException as e:
        print(f"    [API] ❌ FALHA: {e}")
        return False

def save_artifact(task_id, content, filename):
    try:
        task_dir = os.path.join(ARTIFACTS_DIR, f"TSK_{task_id}")
        os.makedirs(task_dir, exist_ok=True)
        filepath = os.path.join(task_dir, filename)
        with open(filepath, 'w', encoding='utf-8') as f: f.write(content)
        print(f"    [FS] Artefato '{filename}' salvo em '{filepath}'.")
        return filepath
    except IOError as e: 
        print(f"    [FS] ❌ Erro ao salvar artefato: {e}")
        return None

def read_artifact(task_id, filename):
    try:
        filepath = os.path.join(ARTIFACTS_DIR, f"TSK_{task_id}", filename)
        with open(filepath, 'r', encoding='utf-8') as f:
            return f.read()
    except (IOError, FileNotFoundError): return None

# --- Lógica do Worker (Corrigida) ---

def generate_detailed_op_report(task_data):
    task_id = task_data['task_id']
    print(f"    [Worker-{task_id}] Gerando Relatório Operacional detalhado via Oráculo...")
    context_blocks = ""
    for block in sorted(task_data.get('blocks', []), key=lambda b: b.get('sequence', 0)):
        context_blocks += f"--- Bloco (Autor: {block.get('author_id', 'N/A')}, Tipo: {block.get('block_type', 'N/A')}) ---\n{block.get('content', 'N/A')}\n\n"
    prompt = f"""
    **PERSONA:** Você é o Engenheiro de Sistemas do SOFIA.
    **MISSÃO:** Crie um "Relatório Operacional" em Markdown para a Tarefa TSK_{task_id}, narrando sua jornada pelo sistema. Descreva cada passo: a criação pelo Executor, o enriquecimento pelo Escriba, a análise pelo RH e a execução final pelo Worker.
    **CONTEXTO COMPLETO DA TAREFA (BLOCOS EM ORDEM):**
    {context_blocks}
    ---
    **FORMATO DE SAÍDA OBRIGATÓRIO:**
    Responda APENAS com um objeto JSON válido contendo uma única chave "op_report_content". O valor deve ser o relatório completo em Markdown.
    """
    response = chamar_oraculo_json(prompt, model=GROQ_MODEL_FAST)
    op_report_content = response.get('op_report_content', f"# Relatório Operacional TSK_{task_id}\n\nERRO: A LLM não conseguiu gerar o relatório detalhado.")
    if "ERRO:" in op_report_content:
        op_report_content += f"\nResposta bruta da LLM: {json.dumps(response, indent=2)}"
    print(f"    [Worker-{task_id}] ✅ Relatório Operacional detalhado gerado.")
    return op_report_content

def execute_atomic_mission(session, task_data):
    task_id = task_data['task_id']
    print(f"[Worker-{task_id}] Executando missão atômica.")
    
    # <<< CORREÇÃO APLICADA AQUI >>>
    # Pega AMBOS os contextos: o Gênesis (com os links) e o Enriquecido.
    genesis_block = next((b['content'] for b in task_data.get('blocks', []) if b.get('block_type') == 'GenesisPrompt'), '')
    enriched_block = next((b['content'] for b in task_data.get('blocks', []) if b.get('block_type') == 'EnrichedDescription'), 'N/A')
    
    if enriched_block == 'N/A': 
        print(f"[Worker-{task_id}] ❌ ERRO: Bloco 'EnrichedDescription' não encontrado.")
        update_task_status(session, task_id, 5)
        return False

    # Combina os dois contextos para o prompt final.
    full_context_prompt = f"""
    {enriched_block}

    {genesis_block}
    """

    prompt = f"""
    **PERSONA:** Você é um analista de pesquisa sênior e especialista técnico.
    **MISSÃO:** Desenvolva um relatório detalhado e aprofundado em Markdown sobre o tópico abaixo. Sua análise DEVE se basear primariamente nas fontes de conhecimento listadas no "Contexto de Pesquisa Permanente". Não invente informações; use os repositórios como sua principal fonte da verdade.
    
    **TÓPICO E CONTEXTO PARA O RELATÓRIO:**
    ---
    {full_context_prompt}
    ---
    
    **FORMATO DE SAÍDA OBRIGATÓRIO:**
    Sua resposta DEVE ser um objeto JSON válido. Este JSON DEVE conter uma única chave chamada "report_content". O valor desta chave DEVE ser o relatório completo e detalhado em Markdown.
    """
    
    print(f"\n[Worker-{task_id}] --- Gerando FinalReport via Oráculo (com contexto completo) ---")
    response = chamar_oraculo_json(prompt, model=GROQ_MODEL_POWERFUL)
    final_report_content = response.get('report_content', f"ERRO: A LLM não retornou a chave 'report_content'.")
    if "ERRO:" in final_report_content:
        final_report_content += f"\nResposta bruta da LLM: {json.dumps(response, indent=2)}"
    save_artifact(task_id, final_report_content, "FinalReport.md")
    
    print(f"[Worker-{task_id}] ✅ FinalReport gerado. Atualizando status para 52.")
    if not update_task_status(session, task_id, 52): return False
    
    op_report_content = generate_detailed_op_report(task_data)
    save_artifact(task_id, op_report_content, "OperationalReport.md")
    
    print(f"[Worker-{task_id}] ✅ OperationalReport detalhado gerado. Atualizando status para 3.")
    update_task_status(session, task_id, 3)
    return True

def execute_supervisor_mission(session, task_data):
    # Esta função não será chamada com a correção do RH, mas a deixamos corrigida.
    parent_task_id = task_data['task_id']
    print(f"[Supervisor-{parent_task_id}] Iniciando protocolo de supervisão.")
    # ... (código da função original para compilar relatórios) ...
    
    # <<< CORREÇÃO APLICADA AQUI >>>
    # Garante que o relatório operacional seja gerado também para a tarefa-mãe.
    op_report_content = generate_detailed_op_report(task_data)
    save_artifact(parent_task_id, op_report_content, "OperationalReport.md")
    
    print(f"[Supervisor-{parent_task_id}] ✅ Missão de compilação concluída. Atualizando status para 3.")
    update_task_status(session, parent_task_id, 3)
    return True

if __name__ == "__main__":
    agent_instance_id = f"WorkerAgent-{random.randint(1000, 9999)}"
    print(f"--- 🚀 {agent_instance_id} iniciado. ('O Orquestrador Autônomo' v6.1.0 - CORRIGIDO) ---")
    with requests.Session() as session:
        while True:
            try:
                task = find_and_claim_worker_task(session)
                if task:
                    task_id = task['task_id']
                    # Com a correção do RH, este check é redundante, mas seguro.
                    is_parent = any(t.get('parent_task_id') == task_id for t in requests.get(f"{SOFIA_API_BASE_URL}/tasks/").json())
                    if is_parent:
                        execute_supervisor_mission(session, task)
                    else:
                        execute_atomic_mission(session, task)
                else:
                    time.sleep(10)
            except KeyboardInterrupt:
                print(f"\n--- 🛑 {agent_instance_id} desligando... ---")
                break
            except Exception as e:
                print(f"\n--- 🚨 Erro inesperado no loop principal: {e} ---")
                time.sleep(15)

def execute_supervisor_mission(session, task_data):

def execute_supervisor_mission(session, task_data):

            if not child_tasks:
                print(f"    [Supervisor-{parent_task_id}] AVISO: Nenhuma tarefa-filha encontrada. Prosseguindo para compilação.")
                break

            completed_children = [child for child in child_tasks if child.get('status', {}).get('status_id') == 3]
            
            print(f"    [Supervisor-{parent_task_id}] Progresso: {len(completed_children)} de {len(child_tasks)} filhas concluídas.")

                break
            
            time.sleep(15)
        except Exception as e:
            print(f"    [Supervisor-{parent_task_id}] ❌ Erro durante o monitoramento: {e}. Tentando novamente em 20s.")
            time.sleep(20)

    child_reports_context = ""
    for child in child_tasks:

            child_reports_context += f"\n\n---\n\n**RELATÓRIO DA SUB-TAREFA TSK_{child_id}**\n\n{report_content}"
        else:
            child_reports_context += f"\n\n---\n\n**RELATÓRIO DA SUB-TAREFA TSK_{child_id}**\n\n*ERRO: Não foi possível ler o artefato.*"

    # Agora é seguro usar 'blocks' porque full_parent_task_data os contém.
    parent_description = next((b['content'] for b in full_parent_task_data.get('blocks', []) if b.get('block_type') == 'EnrichedDescription'), 'N/A')
    
    synthesis_prompt = f"""
    **PERSONA:** Você é um Editor-Chefe...
    **MISSÃO PRINCIPAL:**\n---\n{parent_description}\n---\n
    **RELATÓRIOS COMPLETOS DOS ESPECIALISTAS:**\n---\n{child_reports_context}\n---\n
    **SUA TAREFA FINAL:** ...
    **FORMATO DE SAÍDA OBRIGATÓRIO:** ... "synthesis_report" ...
    """

    print(f"\n[Supervisor-{parent_task_id}] --- Gerando Dossiê Final via Oráculo ---")
    response = chamar_oraculo_json(prompt=synthesis_prompt, model=GROQ_MODEL_POWERFUL)
    final_report_content = response.get('synthesis_report', f"ERRO: A LLM não retornou a chave 'synthesis_report'.")
    if "ERRO:" in final_report_content:
        final_report_content += f"\nResposta bruta da LLM: {json.dumps(response, indent=2)}"
    save_artifact(parent_task_id, final_report_content, "FinalReport.md")

    # Passo 4: Gerar o relatório operacional detalhado e finalizar.

    save_artifact(parent_task_id, op_report_content, "OperationalReport.md")

    update_task_status(session, parent_task_id, 3)
    return True

# --- Funções de API (sem alterações) ---
def find_and_claim_worker_task(session):
    try:
        response = session.get(f"{SOFIA_API_BASE_URL}/tasks/")
        if response.status_code == 200:
            for task in response.json():
                if task.get('status', {}).get('status_id') == 1:
                    task_id = task['task_id']
                    print(f"    [Worker-{task_id}] Tarefa com status 1 encontrada. Reivindicando (status 50)...")
                    claim_response = session.patch(f"{SOFIA_API_BASE_URL}/tasks/{task_id}", json={"status_id": 50})
                    if claim_response.status_code == 200:
                        return task
        return None
    except requests.exceptions.RequestException:
        return None

def update_task_status(session, task_id, new_status_id):
    try:
        print(f"    [API] Tentando atualizar task {task_id} para status {new_status_id}...")
        response = session.patch(f"{SOFIA_API_BASE_URL}/tasks/{task_id}", json={"status_id": new_status_id})
        response.raise_for_status()
        print(f"    [API] ✅ Sucesso.")
        return True
    except requests.exceptions.RequestException as e:
        print(f"    [API] ❌ FALHA: {e}")
        return False

def save_artifact(task_id, content, filename):
    try:
        task_dir = os.path.join(ARTIFACTS_DIR, f"TSK_{task_id}")
        os.makedirs(task_dir, exist_ok=True)
        filepath = os.path.join(task_dir, filename)
        with open(filepath, 'w', encoding='utf-8') as f: f.write(content)
        print(f"    [FS] Artefato '{filename}' salvo em '{filepath}'.")
        return filepath
    except IOError as e: 
        print(f"    [FS] ❌ Erro ao salvar artefato: {e}")
        return None

def read_artifact(task_id, filename):
    try:
        filepath = os.path.join(ARTIFACTS_DIR, f"TSK_{task_id}", filename)
        with open(filepath, 'r', encoding='utf-8') as f:
            return f.read()
    except (IOError, FileNotFoundError): return None

    prompt = f"""
    **PERSONA:** Você é o Engenheiro de Sistemas do SOFIA.

    **CONTEXTO COMPLETO DA TAREFA (BLOCOS EM ORDEM):**
    {context_blocks}
    ---
    **FORMATO DE SAÍDA OBRIGATÓRIO:**
    Responda APENAS com um objeto JSON válido contendo uma única chave "op_report_content". O valor deve ser o relatório completo em Markdown.
    """
    response = chamar_oraculo_json(prompt, model=GROQ_MODEL_FAST)
    op_report_content = response.get('op_report_content', f"# Relatório Operacional TSK_{task_id}\n\nERRO: A LLM não conseguiu gerar o relatório detalhado.")
    if "ERRO:" in op_report_content:
        op_report_content += f"\nResposta bruta da LLM: {json.dumps(response, indent=2)}"
    print(f"    [Worker-{task_id}] ✅ Relatório Operacional detalhado gerado.")
    return op_report_content
