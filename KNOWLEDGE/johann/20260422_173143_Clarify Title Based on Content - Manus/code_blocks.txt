    # O fluxo normal continua apenas se a pergunta for válida.

    stabilized_question = stabilize_query(user_question, groq_client)

    print("--- ✅ FLUXO COMPLETO CONCLUÍDO COM SUCESSO ---")
    return {"status": "success", "artifact_path": final_artifact_path}
