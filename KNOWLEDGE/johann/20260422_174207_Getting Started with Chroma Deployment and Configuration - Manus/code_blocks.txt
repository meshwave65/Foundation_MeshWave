import os
import shutil
from fastapi import FastAPI, HTTPException, UploadFile, File, Form, BackgroundTasks
from fastapi.middleware.cors import CORSMiddleware
from fastapi.responses import PlainTextResponse
from datetime import datetime
from architecture import ARTIFACTS_DIR, MONITOR_DIR

# Importa todos os routers que fazem parte do seu projeto
from backend.app.routers import (
    tasks_router, 
    roadmap_router, 
    oracle_router, 
    topics_router, 
    interaction_router,
    messages_router,
    missions_router,
    workspace_router  # <-- 1. IMPORTAÇÃO DO NOVO ROUTER
)

from backend.app.database.connect_db import engine, SessionLocal, Base
from backend.app.models import models

# Configurar as tabelas
Base.metadata.create_all(bind=engine)

app = FastAPI(
    title="SOFIA - Strategic Operations Framework and Intelligent Automation",
    description="API para o ecossistema SOFIA.",
    version="2.8.3"  # Versão atualizada
)

app.add_middleware(
    CORSMiddleware,
    allow_origins=["*"],
    allow_credentials=True,
    allow_methods=["*"],
    allow_headers=["*"],
)

# Inclui todos os routers do projeto
app.include_router(tasks_router.router)
app.include_router(roadmap_router.router)
app.include_router(oracle_router.router)
app.include_router(topics_router.router)
app.include_router(interaction_router.router)
app.include_router(messages_router.router)
app.include_router(missions_router.router)
app.include_router(workspace_router.router) # <-- 2. INCLUSÃO DO NOVO ROUTER NA APLICAÇÃO

@app.get("/", response_class=PlainTextResponse)
async def read_root():
    return "SOFIA API is running."

@app.post("/api/v1/artifacts", tags=["Artifacts"])
async def upload_artifact(
    task_id: int = Form(...),
    file: UploadFile = File(...)
):
    os.makedirs(ARTIFACTS_DIR, exist_ok=True)
    file_path = os.path.join(ARTIFACTS_DIR, file.filename)
    if not os.path.abspath(file_path).startswith(os.path.abspath(ARTIFACTS_DIR)):
        raise HTTPException(status_code=400, detail="Caminho de arquivo inválido.")
    try:
        with open(file_path, "wb") as buffer:
            shutil.copyfileobj(file.file, buffer)
    except Exception as e:
        raise HTTPException(status_code=500, detail=f"Falha ao salvar o arquivo: {str(e)}")
    finally:
        if file:
            file.file.close()

# Endpoint para monitoramento
MONITOR_ARTIFACTS_DIR = MONITOR_DIR
os.makedirs(MONITOR_ARTIFACTS_DIR, exist_ok=True)

@app.post("/api/v1/monitor", tags=["Monitor"])
async def receive_monitor_artifact(
    background_tasks: BackgroundTasks,
    file: UploadFile = File(...),
    artifact_type: str = Form("gps")
):
    if artifact_type not in ["gps", "audio", "image"]:
        raise HTTPException(status_code=400, detail="Tipo de artefato inválido. Use: 'gps', 'audio', 'image'")

    try:
        contents = await file.read()
        
        type_config = {
            "gps": {"prefix": "gps", "ext": "json"},
            "audio": {"prefix": "audio", "ext": "mp3"}, 
            "image": {"prefix": "img", "ext": "jpg"}
        }
        config = type_config[artifact_type]
        
        timestamp = datetime.now().strftime("%Y%m%d_%H%M%S_%f")
        filename = f"{config['prefix']}_{timestamp}.{config['ext']}"
        file_path = os.path.join(MONITOR_ARTIFACTS_DIR, filename)
        
        def save_file():
            with open(file_path, "wb") as f:
                f.write(contents)
            print(f"Artefato de monitoramento salvo: {file_path}")
        
        background_tasks.add_task(save_file)
        
        return {
            "status": "success",
            "message": "Artefato recebido e está sendo processado.",
            "artifact_type": artifact_type,
            "filename": filename,
            "received_at": datetime.now().isoformat()
        }
    except Exception as e:
        raise HTTPException(status_code=500, detail=f"Erro ao processar artefato: {str(e)}")
    finally:
        await file.close()
