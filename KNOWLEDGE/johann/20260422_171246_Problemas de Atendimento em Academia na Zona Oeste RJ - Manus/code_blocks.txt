    <script>
        let currentLevel = 0;

        function updateCurrentStep(description) {
            const stepElement = document.getElementById('currentStep');
            const descElement = document.getElementById('stepDescription');
            stepElement.style.display = 'block';
            descElement.innerHTML = description;
        }

        function highlightNode(nodeId) {
            document.querySelectorAll('.node').forEach(node => node.classList.remove('active'));
            const node = document.getElementById(nodeId);
            if (node) {
                node.classList.add('active');
                node.scrollIntoView({ behavior: 'smooth', block: 'center' });
            }
        }

        function simulateScenario(scenario) {
            switch(scenario) {
                case 'positive':
                    highlightNode('positive');
                    updateCurrentStep('<strong>Cenário:</strong> Conduta exemplar identificada.  
<strong>Ação:</strong> Aplicar reforço positivo, elogiando o colaborador.');
                    currentLevel = 0;
                    break;
                case 'light':
                    highlightNode('level1');
                    updateCurrentStep('<strong>Cenário:</strong> Falha leve detectada (Ex: no celular).  
<strong>Ação:</strong> Aplicar correção imediata (Nível 1).');
                    currentLevel = 1;
                    break;
                case 'serious':
                    if (currentLevel < 2) {
                        highlightNode('level2');
                        updateCurrentStep('<strong>Cenário:</strong> Falha grave ou reincidência de falha leve.  
<strong>Ação:</strong> Aplicar advertência verbal formal (Nível 2).');
                        currentLevel = 2;
                    } else if (currentLevel === 2) {
                        highlightNode('level3');
                        updateCurrentStep('<strong>Cenário:</strong> Nova reincidência após advertência verbal.  
<strong>Ação:</strong> Aplicar advertência por escrito (Nível 3).');
                        currentLevel = 3;
                    } else {
                        highlightNode('level4');
                        updateCurrentStep('<strong>Cenário:</strong> Múltiplas reincidências.  
<strong>Ação:</strong> Iniciar processo de afastamento (Nível 4).');
                        currentLevel = 4;
                    }
                    break;
                case 'critical':
                    highlightNode('level4');
                    updateCurrentStep('<strong>Cenário:</strong> Falha gravíssima detectada (Ex: assédio).  
<strong>Ação:</strong> Afastamento imediato e acionamento jurídico (Nível 4/5).');
                    currentLevel = 5;
                    break;
            }
        }

        function resetSimulation() {
            document.querySelectorAll('.node').forEach(node => node.classList.remove('active'));
            document.getElementById('currentStep').style.display = 'none';
            currentLevel = 0;
            highlightNode('start');
        }

        document.addEventListener('DOMContentLoaded', () => highlightNode('start'));
    </script>
