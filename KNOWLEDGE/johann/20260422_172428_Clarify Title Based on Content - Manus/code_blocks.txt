# /home/mesh/home/sofia/engine/condenser_service.py
# VERSÃO 1.0.0 - Serviço dedicado para a PoC do Condenser Agent

import sys
from pathlib import Path
from fastapi import FastAPI
from fastapi.middleware.cors import CORSMiddleware
from pydantic import BaseModel
from groq import Groq

# Adiciona o caminho do engine para importar a arquitetura
ENGINE_ROOT = Path(__file__).resolve().parent
if str(ENGINE_ROOT) not in sys.path:
    sys.path.append(str(ENGINE_ROOT))

# Importa as configurações da fonte da verdade
try:
    from architecture import GROQ_API_KEY, CONDENSER_FRONTEND_PORT
except ImportError as e:
    print(f"❌ ERRO CRÍTICO: Não foi possível importar de 'architecture.py'. {e}")
    sys.exit(1)

# --- Modelo de Dados para a Requisição ---
class CondenseRequest(BaseModel):
    text: str

# --- Inicialização do FastAPI ---
app = FastAPI(
    title="SOFIA - Condenser Service (PoC)",
    description="Um serviço dedicado para demonstrar a capacidade de condensação de texto do SOFIA.",
    version="1.0.0"
)

# --- Configuração do CORS ---
# Permite que o frontend (rodando na porta 5175) se comunique com este backend.
origins = [
    f"http://localhost:{CONDENSER_FRONTEND_PORT}",
    "http://localhost" # Adicionado para flexibilidade
]

app.add_middleware(
    CORSMiddleware,
    allow_origins=origins,
    allow_credentials=True,
    allow_methods=["*"],
    allow_headers=["*"],
 )

# --- Inicialização do Cliente da API ---
try:
    groq_client = Groq(api_key=GROQ_API_KEY)
except Exception as e:
    print(f"❌ ERRO CRÍTICO: Falha ao inicializar o cliente Groq. Verifique a GROQ_API_KEY. {e}")
    groq_client = None

# --- Prompt de Condensação ---
PROMPT_CONDENSAR = """
Condense o texto a seguir de forma brutal, removendo toda e qualquer palavra ou frase que não seja absolutamente essencial para o entendimento do núcleo da informação. Seu objetivo é a máxima redução de tokens, mantendo a integridade semântica. Não adicione introduções, conclusões ou qualquer texto explicativo. Retorne apenas o texto condensado.

Texto Original:
---
{text}
---
Texto Condensado:
"""

# --- O Endpoint que Faltava ---
@app.post("/condense")
async def condense_text(request: CondenseRequest):
    """
    Recebe um texto e retorna sua versão condensada.
    """
    if not groq_client:
        return {"error": "Serviço de IA não inicializado."}
    if not request.text.strip():
        return {"error": "O texto não pode estar vazio."}

    try:
        completion = groq_client.chat.completions.create(
            messages=[
                {
                    "role": "user",
                    "content": PROMPT_CONDENSAR.format(text=request.text),
                }
            ],
            model="llama3-8b-8192", # Usando um modelo rápido para a PoC
            temperature=0.1,
        )
        condensed_result = completion.choices[0].message.content
        return {"condensed_text": condensed_result}
    except Exception as e:
        return {"error": f"Erro ao comunicar com a API da Groq: {str(e)}"}

print("✅ Serviço de Condensação (condenser_service.py) pronto para ser iniciado.")
