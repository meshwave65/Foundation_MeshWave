class BayesianModule {
    private val prob_GoodLatency_Given_Legit = 0.8f
    private val prob_GoodLatency_Given_Impostor = 0.3f

    fun calculateConfidence(evidence: EvidenceBundle): ConfidenceScore {
        println("[Bayes]: Calculando confiança com base em evidências resilientes.")

        if (evidence.validationResult.status == ValidationStatus.INVALID) {
            println("[Bayes]: Evidência conclusiva: Segmento inválido. Confiança = 0%.")
            return ConfidenceScore(0.0f)
        }

        val priorLegit = when (evidence.syncResult.status) {
            SyncStatus.SUCCESS -> 0.7f
            SyncStatus.UNSTABLE -> 0.4f
            SyncStatus.FAILED -> 0.2f
        }
        println("[Bayes]: Crença Anterior (baseada na Sincronização): ${"%.0f".format(priorLegit * 100)}%")

        var likelihoodFactor = 1.0f // Fator neutro por padrão

        // A MÁGICA ACONTECE AQUI:
        // Verificamos o TIPO do valor, não apenas o valor em si.
        when (evidence.syncResult.averageLatencyMs) {
            is ResilientValue.Real -> {
                // Temos um dado real, podemos usá-lo para o cálculo.
                val realSyncLatency = evidence.syncResult.averageLatencyMs.value
                if (evidence.authLatencyMs != null) {
                    val latencyIsGood = evidence.authLatencyMs < (realSyncLatency + 40)
                    val likelihoodGivenLegit = if (latencyIsGood) prob_GoodLatency_Given_Legit else (1 - prob_GoodLatency_Given_Legit)
                    val likelihoodGivenImpostor = if (latencyIsGood) prob_GoodLatency_Given_Impostor else (1 - prob_GoodLatency_Given_Impostor)
                    
                    if (likelihoodGivenImpostor > 0) {
                        likelihoodFactor = likelihoodGivenLegit / likelihoodGivenImpostor
                    }
                    println("[Bayes]: Fator de Verossimilhança (Latência REAL): ${"%.2f".format(likelihoodFactor)}")
                }
            }
            is ResilientValue.Placeholder -> {
                // Temos um placeholder. A latência da sincronização não é confiável.
                // Não podemos calcular a verossimilhança da latência.
                // O fator permanece neutro (1.0), e a baixa confiança virá do `priorLegit` já reduzido.
                println("[Bayes]: Fator de Verossimilhança (Latência): IGNORADO (dado é placeholder).")
            }
        }

        val priorOdds = priorLegit / (1 - priorLegit)
        val posteriorOdds = likelihoodFactor * priorOdds
        val finalProbability = (posteriorOdds / (1 + posteriorOdds)).coerceIn(0f, 1f)

        println("[Bayes]: Análise concluída. Confiança final: ${"%.2f".format(finalProbability * 100)}%")
        return ConfidenceScore(finalProbability)
    }
}

class TimeSyncManager(private val device: Device) {
    private var observedTimestamps = mutableListOf<Pair<Long, Long>>()

    suspend fun startServerProcess(connection: FakeConnection) {
        observedTimestamps.clear()
        println("[${device.name}-Sync]: Iniciando sincronização.")
        repeat(TimeSyncProtocol.SYNC_PACKET_COUNT) {
            connection.send(device, SyncPacket(device.clock.time()))
        }
    }

    suspend fun handleClientPacket(packet: SyncPacket, connection: FakeConnection) {
        observedTimestamps.add(Pair(packet.originTimestamp, device.clock.time()))

        if (observedTimestamps.size == TimeSyncProtocol.SYNC_PACKET_COUNT) {
            val latencies = observedTimestamps.zipWithNext { a, b -> (b.second - a.second) - (b.first - a.first) }
            val oneWayApproximation = (latencies.average().toLong() / 2).coerceAtLeast(0)
            
            val t3 = device.clock.time()
            val t4_predicted = t3 + oneWayApproximation
            connection.send(device, CalibrationPacket(t3, t4_predicted))
            observedTimestamps.clear()
        }
    }
}

enum class ValidationStatus { VALID, INVALID }
data class ValidationResult(val status: ValidationStatus)
data class ArcChallenge(val segmentIndex: Int, val segmentLength: Int, val timestamp: Long)
data class ArcResponse(val segment: String)

object ArcProtocol {
    private fun sha256(input: String): String = MessageDigest.getInstance("SHA-256")
        .digest(input.toByteArray()).fold("") { str, it -> str + "%02x".format(it) }

    fun createSegment(secretId: String, challenge: ArcChallenge): String {
        val baseHash = sha256(secretId)
        val contextualHash = sha256(baseHash + challenge.timestamp)
        return contextualHash.substring(challenge.segmentIndex, challenge.segmentIndex + challenge.segmentLength)
    }

    fun validate(response: ArcResponse, challenge: ArcChallenge, peerSecretId: String): ValidationResult {
        val expectedSegment = createSegment(peerSecretId, challenge)
        val isValid = response.segment == expectedSegment
        println("[ARC-Protocol]: Validando. Esperado: $expectedSegment, Recebido: ${response.segment}. Válido: $isValid")
        return ValidationResult(if (isValid) ValidationStatus.VALID else ValidationStatus.INVALID)
    }
}

class ArcManager(private val device: Device) {
    fun createChallenge(): ArcChallenge = ArcChallenge(10, 8, device.clock.time())

    fun createResponse(challenge: ArcChallenge): ArcResponse {
        val segment = ArcProtocol.createSegment(device.secretId, challenge)
        println("[${device.name}-ARC]: Respondendo com o segmento: $segment")
        return ArcResponse(segment)
    }
}

class DeviceClock(private val offsetMs: Long = 0) { fun time(): Long = System.currentTimeMillis() + offsetMs }
class FakeConnection(private val deviceA: Device, private val deviceB: Device, private val baseLatencyMs: Long, private val jitterMs: Int) {
    suspend fun send(from: Device, message: Any) {
        val latency = baseLatencyMs + Random.nextInt(-jitterMs, jitterMs)
        delay(latency)
        val receiver = if (from == deviceA) deviceB else deviceA
        receiver.receive(message)
    }
}
data class SyncPacket(val originTimestamp: Long)
data class CalibrationPacket(val originTimestamp: Long, val predictedArrival: Long)

class Device(
    val name: String,
    val clock: DeviceClock,
    val secretId: String,
    var knownPeers: Map<String, String> = emptyMap()
) {
    private val timeSyncManager = TimeSyncManager(this)
    private val arcManager = ArcManager(this)
    private val bayesianModule = BayesianModule()

    private var pendingChallenge: ArcChallenge? = null
    private var authStartTime: Long = 0
    private lateinit var lastSyncResult: SyncResult
    private lateinit var connection: FakeConnection

    fun setConnection(conn: FakeConnection) { connection = conn }

    suspend fun startProcess() { timeSyncManager.startServerProcess(connection) }

    suspend fun receive(message: Any) {
        println("[$name]: Mensagem recebida: ${message::class.simpleName}")
        when (message) {
            is SyncPacket -> timeSyncManager.handleClientPacket(message, connection)
            is CalibrationPacket -> {
                lastSyncResult = TimeSyncProtocol.analyzeCalibration(message, clock.time())
                println("[$name]: Resultado da Sincronização: ${lastSyncResult.status} com latência ${lastSyncResult.averageLatencyMs}")

                val challenge = arcManager.createChallenge()
                pendingChallenge = challenge
                authStartTime = clock.time()
                connection.send(this, challenge)
            }
            is ArcChallenge -> {
                val response = arcManager.createResponse(message)
                connection.send(this, response)
            }
            is ArcResponse -> {
                val authLatency = clock.time() - authStartTime
                val peerName = if (name == "DispositivoA") "DispositivoB" else "DispositivoA"
                val validation = ArcProtocol.validate(message, pendingChallenge!!, knownPeers[peerName]!!)
                
                val evidence = EvidenceBundle(validation, authLatency, lastSyncResult)
                val confidence = bayesianModule.calculateConfidence(evidence)

                if (confidence.probability > 0.85f) {
                    println("\n>>> [$name]: AUTENTICAÇÃO BEM-SUCEDIDA (Confiança: ${"%.1f".format(confidence.probability * 100)}%) <<<\n")
                } else {
                    println("\n>>> [$name]: FALHA NA AUTENTICAÇÃO (Confiança: ${"%.1f".format(confidence.probability * 100)}%) <<<\n")
                }
            }
        }
    }
}

    println("--- INICIANDO SIMULAÇÃO DE AUTENTICAÇÃO RESILIENTE ---")

    val clockA = DeviceClock()
    val clockB = DeviceClock(offsetMs = -80)
    
    val deviceA = Device("DispositivoA", clockA, "segredo-do-A-123")
    val deviceB = Device("DispositivoB", clockB, "segredo-do-B-456")

    deviceA.knownPeers = mapOf("DispositivoB" to deviceB.secretId)
    deviceB.knownPeers = mapOf("DispositivoA" to deviceA.secretId)

    // Para testar o cenário de falha, aumente o jitter para que a previsão erre mais
    val connection = FakeConnection(deviceA, deviceB, baseLatencyMs = 30, jitterMs = 40)
    deviceA.setConnection(connection)
    deviceB.setConnection(connection)
    
    launch {
        deviceB.startProcess()
    }
}

    private val prob_GoodLatency_Given_Impostor = 0.3f

    fun calculateConfidence(evidence: EvidenceBundle): ConfidenceScore {
        println("[Bayes]: Calculando confiança com base em evidências resilientes.")

        if (evidence.validationResult.status == ValidationStatus.INVALID) {
            println("[Bayes]: Evidência conclusiva: Segmento inválido. Confiança = 0%.")
            return ConfidenceScore(0.0f)
        }

        val priorLegit = when (evidence.syncResult.status) {
            SyncStatus.SUCCESS -> 0.7f
            SyncStatus.UNSTABLE -> 0.4f
            SyncStatus.FAILED -> 0.2f
        }
        println("[Bayes]: Crença Anterior (baseada na Sincronização): ${"%.0f".format(priorLegit * 100)}%")

        var likelihoodFactor = 1.0f

        // A lógica agora verifica o TIPO do valor de latência da sincronização
        when (val syncLatency = evidence.syncResult.averageLatencyMs) {
            is ResilientValue.Real -> {
                if (evidence.authLatencyMs != null) {
                    val latencyIsGood = evidence.authLatencyMs < (syncLatency.value + 40)
                    val likelihoodGivenLegit = if (latencyIsGood) prob_GoodLatency_Given_Legit else (1 - prob_GoodLatency_Given_Legit)
                    val likelihoodGivenImpostor = if (latencyIsGood) prob_GoodLatency_Given_Impostor else (1 - prob_GoodLatency_Given_Impostor)
                    
                    if (likelihoodGivenImpostor > 0) {
                        likelihoodFactor = likelihoodGivenLegit / likelihoodGivenImpostor
                    }
                    println("[Bayes]: Fator de Verossimilhança (Latência REAL): ${"%.2f".format(likelihoodFactor)}")
                }
            }
            is ResilientValue.Placeholder -> {
                println("[Bayes]: Fator de Verossimilhança (Latência): IGNORADO (dado é placeholder).")
            }
        }

        val priorOdds = priorLegit / (1 - priorLegit)
        val posteriorOdds = likelihoodFactor * priorOdds
        val finalProbability = (posteriorOdds / (1 + posteriorOdds)).coerceIn(0f, 1f)

        println("[Bayes]: Análise concluída. Confiança final: ${"%.2f".format(finalProbability * 100)}%")
        return ConfidenceScore(finalProbability)
    }
}

class FakeConnection(private val deviceA: Device, private val deviceB: Device, private val baseLatencyMs: Long, private val jitterMs: Int) {
    suspend fun send(from: Device, message: Any) {
        val latency = baseLatencyMs + Random.nextInt(-jitterMs, jitterMs)
        delay(latency)
        val receiver = if (from == deviceA) deviceB else deviceA
        receiver.receive(message)
    }

data class SyncPacket(val originTimestamp: Long)
data class CalibrationPacket(val originTimestamp: Long, val predictedArrival: Long)

class Device(
    val name: String,
    val clock: DeviceClock,
    val secretId: String,
    var knownPeers: Map<String, String> = emptyMap()
) {
    private val timeSyncManager = TimeSyncManager(this)
    private val arcManager = ArcManager(this)
    private val bayesianModule = BayesianModule()

    private var pendingChallenge: ArcChallenge? = null
    private var authStartTime: Long = 0
    private lateinit var lastSyncResult: SyncResult
    private lateinit var connection: FakeConnection

    fun setConnection(conn: FakeConnection) { connection = conn }

    suspend fun startProcess() { timeSyncManager.startServerProcess(connection) }

    suspend fun receive(message: Any) {
        println("[$name]: Mensagem recebida: ${message::class.simpleName}")
        when (message) {
            is SyncPacket -> timeSyncManager.handleClientPacket(message, connection)
            is CalibrationPacket -> {
                lastSyncResult = TimeSyncProtocol.analyzeCalibration(message, clock.time())
                println("[$name]: Resultado da Sincronização: ${lastSyncResult.status} com latência ${lastSyncResult.averageLatencyMs}")

                val challenge = arcManager.createChallenge()
                pendingChallenge = challenge
                authStartTime = clock.time()
                connection.send(this, challenge)
            }
            is ArcChallenge -> {
                val response = arcManager.createResponse(message)
                connection.send(this, response)
            }
            is ArcResponse -> {
                val authLatency = clock.time() - authStartTime
                val peerName = if (name == "DispositivoA") "DispositivoB" else "DispositivoA"
                val validation = ArcProtocol.validate(message, pendingChallenge!!, knownPeers[peerName]!!)
                
                val evidence = EvidenceBundle(validation, authLatency, lastSyncResult)
                val confidence = bayesianModule.calculateConfidence(evidence)

                if (confidence.probability > 0.85f) {
                    println("\n>>> [$name]: AUTENTICAÇÃO BEM-SUCEDIDA (Confiança: ${"%.1f".format(confidence.probability * 100)}%) <<<\n")
                } else {
                    println("\n>>> [$name]: FALHA NA AUTENTICAÇÃO (Confiança: ${"%.1f".format(confidence.probability * 100)}%) <<<\n")
                }
            }
        }
    }
}

fun main( ) {
    embeddedServer(Netty, port = 8080, host = "0.0.0.0") {
        install(WebSockets) {
            pingPeriod = Duration.ofSeconds(15)
            timeout = Duration.ofSeconds(15)
            maxFrameSize = Long.MAX_VALUE
            masking = false
        }

        routing {
            // Endpoint WebSocket que simula a rede
            webSocket("/ws/network") {
                val jitter = call.request.queryParameters["jitter"]?.toIntOrNull() ?: 15
                val latency = call.request.queryParameters["latency"]?.toLongOrNull() ?: 30

                for (frame in incoming) {
                    if (frame is Frame.Text) {
                        // Simula a latência e o jitter da rede
                        val delayMs = latency + Random.nextInt(-jitter, jitter)
                        launch {
                            delay(delayMs)
                            // Retransmite a mensagem para todos os clientes conectados (os 2 dispositivos)
                            send(frame.readText())
                        }
                    }
                }
            }

            // Serve os arquivos estáticos do frontend
            static("/") {
                resources("static")
                defaultResource("static/index.html")
            }
        }
    }.start(wait = true)
}
