// Função principal corrigida: processDiscoveredPeers
fun processDiscoveredPeers(foundPeers: Map<String, DiscoveredPeer>) {
    val now = System.currentTimeMillis()
    var cacheChanged = false

    foundPeers.values.forEach { newPeer ->
        val existingPeer = situationalCache.discoveredPeers[newPeer.macAddress]
        
        if (existingPeer != null) {
            if (existingPeer.deviceName != newPeer.deviceName || existingPeer.status != 1) {
                existingPeer.deviceName = newPeer.deviceName
                existingPeer.status = 1
                cacheChanged = true
            }
            existingPeer.lastSeenTimestamp = now
        } else {
            situationalCache.discoveredPeers[newPeer.macAddress] = newPeer
            cacheChanged = true
        }
    }
    
    if (cacheChanged) {
        sendCacheUpdate()
    }
}
