class IdentityModule(
    private val context: Context,
    private val uiHandler: Handler
) {

    private var myProfile: NodeProfile? = null
    private var isInitialized = false

    companion object {
        private const val TTL1_STALE_MS = 30 * 1000 // 30 segundos
        private const val TTL2_EXPIRED_MS = 60 * 1000 // 60 segundos
    }

    fun getMyProfile(): NodeProfile? = myProfile

    fun getSituationalCache(): SituationalCache = situationalCache

    @SuppressLint("HardwareIds")

        val androidId = Settings.Secure.getString(context.contentResolver, Settings.Secure.ANDROID_ID)
        return "did:mesh:${androidId}"
    }

        val randomSuffix = Random.nextInt(100, 999)
        return "User-$randomSuffix"
    }

    fun initialize(cpaGeohash: String) {
        if (isInitialized) return
        val did = generateDid()
        myProfile = NodeProfile(
            did = did,
            username = generateUsername(),
            cpaGeohash = cpaGeohash,
            currentClaGeohash = cpaGeohash
        )
        situationalCache.nodes[did] = myProfile!!
        isInitialized = true
        logToUi("[Identity] Módulo inicializado. DID: $did")
        sendCacheUpdate()
    }

    fun initializeWithDefault() {
        if (isInitialized) return
        initialize("g9fail") // Usa o geohash de falha
    }

    fun updateNodeLocation(nodeGeohash: String) {
        if (!isInitialized) {
            initialize(nodeGeohash.take(6))
        } else {
            myProfile?.let {
                it.currentClaGeohash = nodeGeohash.take(6)
                // Se a identidade foi criada com falha, atualiza a CPA de origem na primeira localização bem-sucedida
                if (it.cpaGeohash == "g9fail") {
                    it.cpaGeohash = nodeGeohash.take(6)
                }
            }
            sendCacheUpdate()
        }
    }

        val now = System.currentTimeMillis()
        val currentPeerMacs = situationalCache.discoveredPeers.keys.toMutableSet()

        newPeers.forEach { deviceName ->
            // Extrai o MAC do nome do dispositivo (ex: "moto e5 play_f34b")
            val macAddress = deviceName.substringAfterLast("_", deviceName)
            currentPeerMacs.remove(macAddress)
            val existingPeer = situationalCache.discoveredPeers[macAddress]
            if (existingPeer != null) {
                existingPeer.status = 1
                existingPeer.lastSeenTimestamp = now
            } else {
                situationalCache.discoveredPeers[macAddress] = DiscoveredPeer(macAddress, deviceName)
            }
        }

        currentPeerMacs.forEach { macAddress ->
            situationalCache.discoveredPeers.remove(macAddress)
        }

        sendCacheUpdate()
    }

    fun purgeExpiredPeers() {
        val now = System.currentTimeMillis()
        var changed = false
        situationalCache.discoveredPeers.values.forEach { peer ->
            val age = now - peer.lastSeenTimestamp
            val oldStatus = peer.status
            peer.status = when {
                age > TTL2_EXPIRED_MS -> 0
                age > TTL1_STALE_MS -> 2
                else -> 1
            }
            if (oldStatus != peer.status) {
                changed = true
            }
        }
        if (changed) {
            sendCacheUpdate()
        }

        val msg = uiHandler.obtainMessage(AppConstants.WHAT_CACHE_UPDATE, situationalCache)
        uiHandler.sendMessage(msg)
    }

        uiHandler.sendMessage(msg)
    }

package com.meshwave.core

import android.Manifest
import android.annotation.SuppressLint
import android.content.Context
import android.content.pm.PackageManager
import android.os.Handler
import androidx.core.content.ContextCompat
import ch.hsr.geohash.GeoHash
import com.google.android.gms.location.FusedLocationProviderClient
import com.google.android.gms.location.LocationServices
import com.google.android.gms.location.Priority
import java.util.Timer
import java.util.TimerTask

@SuppressLint("MissingPermission")
class LocationModule(
    private val context: Context,
    private val uiHandler: Handler,
    private val identityModule: IdentityModule
) {
    private lateinit var fusedLocationClient: FusedLocationProviderClient
    private var locationTimer: Timer? = null
    private var lastKnownGoodGeohash: String? = null

    companion object {
        private const val LOCATION_UPDATE_INTERVAL_MS = 15000L
    }

    fun start() {
        fusedLocationClient = LocationServices.getFusedLocationProviderClient(context)
        startLocationUpdates()
    }

    fun stop() {
        locationTimer?.cancel()
        locationTimer = null
    }

    private fun startLocationUpdates() {
        if (locationTimer != null) return
        locationTimer = Timer()
        locationTimer?.scheduleAtFixedRate(object : TimerTask() {
            override fun run() {
                requestLocation()
            }
        }, 0, LOCATION_UPDATE_INTERVAL_MS)
    }

    private fun requestLocation() {
        if (ContextCompat.checkSelfPermission(context, Manifest.permission.ACCESS_FINE_LOCATION) != PackageManager.PERMISSION_GRANTED) {
            handleLocationError("Permissão negada.")
            return
        }

        fusedLocationClient.getCurrentLocation(Priority.PRIORITY_HIGH_ACCURACY, null)
            .addOnSuccessListener { location ->
                if (location != null) {
                    val nodeGeohash = GeoHash.withCharacterPrecision(location.latitude, location.longitude, 9).toBase32()
                    identityModule.updateNodeLocation(nodeGeohash)
                    lastKnownGoodGeohash = nodeGeohash
                    sendToUi(LocationData(nodeGeohash.take(6), LocationStatus.UPDATED))
                } else {
                    handleLocationError("Localização nula.")
                }
            }
            .addOnFailureListener { e ->
                handleLocationError("API error - ${e.message}")
            }
    }

    private fun handleLocationError(reason: String) {
        // CORREÇÃO: Garante a inicialização mesmo em falha
        if (!identityModule.isInitialized()) {
            identityModule.initializeWithDefault()
        }

        if (lastKnownGoodGeohash != null) {
            sendToUi(LocationData(lastKnownGoodGeohash!!.take(6), LocationStatus.STALE))
        } else {
            sendToUi(LocationData("Falha", LocationStatus.FAILED))
        }
    }

    private fun sendToUi(data: LocationData) {
        val msg = uiHandler.obtainMessage(AppConstants.WHAT_LOCATION_UPDATE, data)
        uiHandler.sendMessage(msg)
    }
}
