// v0.3.12-alpha
package com.example.meshwave;

import android.annotation.SuppressLint;
import android.app.Notification;
import android.app.NotificationChannel;
import android.app.NotificationManager;
import android.app.Service;
import android.content.Intent;
import android.net.wifi.p2p.WifiP2pConfig;
import android.net.wifi.p2p.WifiP2pDevice;
import android.net.wifi.p2p.WifiP2pInfo;
import android.net.wifi.p2p.WifiP2pManager;
import android.net.wifi.p2p.nsd.WifiP2pDnsSdServiceInfo;
import android.net.wifi.p2p.nsd.WifiP2pDnsSdServiceRequest;
import android.os.Build;
import android.os.Handler;
import android.os.IBinder;
import android.os.Looper;
import android.util.Log;

import androidx.annotation.Nullable;
import androidx.core.app.NotificationCompat;
import androidx.localbroadcastmanager.content.LocalBroadcastManager;

import java.util.HashMap;
import java.util.Map;
import java.util.function.Consumer;

@SuppressLint("MissingPermission" )
public class MeshWaveService extends Service {
    private static final String TAG = "MeshWaveService";
    private static final String CHANNEL_ID = "MeshWaveChannel";
    private static final int NOTIFICATION_ID = 1;

    public static final String ACTION_START_SERVICE = "com.example.meshwave.START_SERVICE";
    public static final String ACTION_SEND_MESSAGE = "com.example.meshwave.SEND_MESSAGE";
    public static final String ACTION_UPDATE_UI = "com.example.meshwave.UPDATE_UI";
    public static final String ACTION_RECEIVE_MESSAGE = "com.example.meshwave.RECEIVE_MESSAGE";
    public static final String ACTION_CONNECTION_INFO = "com.example.meshwave.CONNECTION_INFO";

    public static final String EXTRA_SESSID = "sessid";
    public static final String EXTRA_USERNAME = "username";
    public static final String EXTRA_MESSAGE = "message";
    public static final String EXTRA_STATUS_TEXT = "status_text";
    public static final String EXTRA_NODE_LIST = "node_list";

    private WifiP2pManager manager;
    private WifiP2pManager.Channel channel;
    private WifiP2pDnsSdServiceRequest serviceRequest;
    private CommunicationThread communicationThread;
    private String currentSessid;
    private String username;

    @Override
    public void onCreate() {
        super.onCreate();
        manager = (WifiP2pManager) getSystemService(WIFI_P2P_SERVICE);
        channel = manager.initialize(this, getMainLooper(), null);
        createNotificationChannel();
    }

    @Override
    public int onStartCommand(Intent intent, int flags, int startId) {
        if (intent != null) {
            String action = intent.getAction();
            if (ACTION_START_SERVICE.equals(action)) {
                currentSessid = intent.getStringExtra(EXTRA_SESSID);
                username = intent.getStringExtra(EXTRA_USERNAME);
                startForeground(NOTIFICATION_ID, createNotification("Inicializando..."));
                startRegistrationAndDiscovery();
            } else if (ACTION_SEND_MESSAGE.equals(action)) {
                String message = intent.getStringExtra(EXTRA_MESSAGE);
                if (communicationThread != null) {
                    communicationThread.write(message.getBytes());
                }
            } else if (ACTION_CONNECTION_INFO.equals(action)) {
                WifiP2pInfo info;
                if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.TIRAMISU) {
                    info = intent.getParcelableExtra("info", WifiP2pInfo.class);
                } else {
                    info = intent.getParcelableExtra("info");
                }
                if (info != null) {
                    handleConnectionChange(info);
                }
            }
        }
        return START_STICKY;
    }

    private void startRegistrationAndDiscovery() {
        stopServiceDiscovery();

        Map<String, String> record = new HashMap<>();
        record.put("user", username);
        record.put("sessid", currentSessid);

        WifiP2pDnsSdServiceInfo serviceInfo = WifiP2pDnsSdServiceInfo.newInstance(MainActivity.SERVICE_INSTANCE, MainActivity.SERVICE_REG_TYPE, record);
        manager.addLocalService(channel, serviceInfo, new WifiP2pManager.ActionListener() {
            @Override
            public void onSuccess() {
                Log.d(TAG, "Serviço anunciado. Aguardando 1s para iniciar descoberta.");
                new Handler(Looper.getMainLooper()).postDelayed(() -> startServiceDiscoveryOnly(), 1000);
            }
            @Override
            public void onFailure(int reason) {
                Log.e(TAG, "Falha ao anunciar serviço: " + reason);
            }
        });
    }

    private void startServiceDiscoveryOnly() {
        WifiP2pManager.DnsSdTxtRecordListener txtListener = (fullDomain, record, device) -> {
            if (record.containsKey("sessid") && record.get("sessid").equals(currentSessid)) {
                Log.d(TAG, "Nó compatível encontrado: " + device.deviceName);
                connectToDevice(device);
            }
        };

        manager.setDnsSdResponseListeners(channel, null, txtListener);
        serviceRequest = WifiP2pDnsSdServiceRequest.newInstance();
        manager.addServiceRequest(channel, serviceRequest, new WifiP2pManager.ActionListener() {
            @Override
            public void onSuccess() {
                manager.discoverServices(channel, new WifiP2pManager.ActionListener() {
                    @Override
                    public void onSuccess() {
                        Log.d(TAG, "Descoberta de serviços iniciada.");
                        broadcastStateUpdate("Procurando...", "Nenhum nó detectado.");
                    }
                    @Override
                    public void onFailure(int reason) {
                        Log.e(TAG, "Falha ao iniciar descoberta: " + reason);
                    }
                });
            }
            @Override
            public void onFailure(int reason) {
                Log.e(TAG, "Falha ao adicionar requisição de serviço: " + reason);
            }
        });
    }

    private void connectToDevice(WifiP2pDevice device) {
        if (device == null) return;
        WifiP2pConfig config = new WifiP2pConfig();
        config.deviceAddress = device.deviceAddress;
        manager.connect(channel, config, new WifiP2pManager.ActionListener() {
            @Override
            public void onSuccess() { Log.d(TAG, "Enviando pedido de conexão para: " + device.deviceName); }
            @Override
            public void onFailure(int reason) { Log.e(TAG, "Falha ao conectar: " + reason); }
        });
    }

    public void handleConnectionChange(WifiP2pInfo info) {
        if (info.groupFormed) {
            stopServiceDiscovery();
            if (communicationThread != null) {
                communicationThread.close();
            }
            Consumer<String> messageConsumer = this::broadcastReceivedMessage;
            communicationThread = new CommunicationThread(info.groupOwnerAddress, info.isGroupOwner, messageConsumer);
            communicationThread.start();
            String status = info.isGroupOwner ? "Conectado (Líder)" : "Conectado (Cliente)";
            broadcastStateUpdate(status, "Conectado a " + (info.isGroupOwner ? "clientes" : "líder"));
        } else {
            broadcastStateUpdate("Desconectado", "Nenhum nó detectado.");
            startRegistrationAndDiscovery();
        }
    }

    private void stopServiceDiscovery() {
        if (manager != null && channel != null) {
            manager.clearLocalServices(channel, null);
            if (serviceRequest != null) {
                manager.removeServiceRequest(channel, serviceRequest, null);
            }
        }
    }

    private void broadcastReceivedMessage(String message) {
        Intent intent = new Intent(ACTION_RECEIVE_MESSAGE);
        intent.putExtra(EXTRA_MESSAGE, message);
        LocalBroadcastManager.getInstance(this).sendBroadcast(intent);
    }

    private void broadcastStateUpdate(String status, String nodeList) {
        Intent intent = new Intent(ACTION_UPDATE_UI);
        intent.putExtra(EXTRA_STATUS_TEXT, status);
        intent.putExtra(EXTRA_NODE_LIST, nodeList);
        LocalBroadcastManager.getInstance(this).sendBroadcast(intent);
    }

    private void createNotificationChannel() {
        if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.O) {
            NotificationChannel serviceChannel = new NotificationChannel(
                    CHANNEL_ID, "MeshWave Service Channel", NotificationManager.IMPORTANCE_DEFAULT);
            getSystemService(NotificationManager.class).createNotificationChannel(serviceChannel);
        }
    }

    private Notification createNotification(String text) {
        return new NotificationCompat.Builder(this, CHANNEL_ID)
                .setContentTitle("MeshWave Ativo")
                .setContentText(text)
                .setSmallIcon(R.drawable.ic_status)
                .build();
    }

    @Override
    public void onDestroy() {
        super.onDestroy();
        if (communicationThread != null) {
            communicationThread.close();
        }
        stopServiceDiscovery();
        if (manager != null && channel != null) {
            manager.removeGroup(channel, null);
        }
    }

    @Nullable
    @Override
    public IBinder onBind(Intent intent) {
        return null;
    }
}

// v0.3.12-alpha
package com.example.meshwave;

import android.Manifest;
import android.annotation.SuppressLint;
import android.content.BroadcastReceiver;
import android.content.Context;
import android.content.Intent;
import android.content.IntentFilter;
import android.content.pm.PackageManager;
import android.location.Location;
import android.net.wifi.p2p.WifiP2pInfo;
import android.net.wifi.p2p.WifiP2pManager;
import android.os.Build;
import android.os.Bundle;
import android.provider.Settings;
import android.util.Log;
import android.widget.Toast;

import androidx.annotation.NonNull;
import androidx.appcompat.app.AppCompatActivity;
import androidx.core.app.ActivityCompat;
import androidx.core.content.ContextCompat;
import androidx.fragment.app.Fragment;
import androidx.fragment.app.FragmentManager;
import androidx.localbroadcastmanager.content.LocalBroadcastManager;

import com.google.android.gms.location.FusedLocationProviderClient;
import com.google.android.gms.location.LocationServices;
import com.google.android.material.bottomnavigation.BottomNavigationView;

import java.util.Random;

public class MainActivity extends AppCompatActivity implements ChatFragmentListener {

    private static final String TAG = "MainActivity_v0.3.12";
    private static final int PERMISSIONS_REQUEST_CODE = 101;
    public static final String SERVICE_INSTANCE = "_meshwave";
    public static final String SERVICE_REG_TYPE = "_meshwave._tcp";

    private FragmentManager fragmentManager;
    private StatusFragment statusFragment;
    private ChatFragment chatFragment;

    private WifiP2pManager manager;
    private WifiP2pManager.Channel channel;
    private BroadcastReceiver p2pReceiver;
    private BroadcastReceiver uiUpdateReceiver;

    private MeshWaveIdentity myIdentity;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        initializeViewsAndFragments(savedInstanceState);
        initializeNetworkLogic();
        setupUiUpdateReceiver();
        checkPrerequisitesAndStart();
    }

    @Override
    protected void onDestroy() {
        super.onDestroy();
        LocalBroadcastManager.getInstance(this).unregisterReceiver(uiUpdateReceiver);
        unregisterReceiver(p2pReceiver);
    }

    @Override
    public void onSendMessage(String message) {
        if (myIdentity == null) {
            Toast.makeText(this, "Identidade ainda não criada.", Toast.LENGTH_SHORT).show();
            return;
        }
        Intent intent = new Intent(this, MeshWaveService.class);
        intent.setAction(MeshWaveService.ACTION_SEND_MESSAGE);
        intent.putExtra(MeshWaveService.EXTRA_MESSAGE, myIdentity.getUsername() + ": " + message);
        startService(intent);
    }

    private void initializeViewsAndFragments(Bundle savedInstanceState) {
        fragmentManager = getSupportFragmentManager();
        statusFragment = new StatusFragment();
        chatFragment = new ChatFragment();

        BottomNavigationView bottomNavigationView = findViewById(R.id.bottom_navigation);
        bottomNavigationView.setOnItemSelectedListener(item -> {
            Fragment selectedFragment = null;
            int itemId = item.getItemId();
            if (itemId == R.id.nav_status) {
                selectedFragment = statusFragment;
            } else if (itemId == R.id.nav_chat) {
                selectedFragment = chatFragment;
            }
            if (selectedFragment != null) {
                fragmentManager.beginTransaction().replace(R.id.fragment_container, selectedFragment).commit();
            }
            return true;
        });

        if (savedInstanceState == null) {
            fragmentManager.beginTransaction().replace(R.id.fragment_container, statusFragment).commit();
        }
    }

    private void initializeNetworkLogic() {
        manager = (WifiP2pManager) getSystemService(Context.WIFI_P2P_SERVICE);
        channel = manager.initialize(this, getMainLooper(), null);
        p2pReceiver = new WiFiDirectBroadcastReceiver(manager, channel, this);
        IntentFilter intentFilter = new IntentFilter();
        intentFilter.addAction(WifiP2pManager.WIFI_P2P_STATE_CHANGED_ACTION);
        intentFilter.addAction(WifiP2pManager.WIFI_P2P_CONNECTION_CHANGED_ACTION);
        registerReceiver(p2pReceiver, intentFilter);
    }

    private void setupUiUpdateReceiver() {
        uiUpdateReceiver = new BroadcastReceiver() {
            @Override
            public void onReceive(Context context, Intent intent) {
                String action = intent.getAction();
                if (MeshWaveService.ACTION_UPDATE_UI.equals(action)) {
                    String status = intent.getStringExtra(MeshWaveService.EXTRA_STATUS_TEXT);
                    String nodeList = intent.getStringExtra(MeshWaveService.EXTRA_NODE_LIST);
                    updateStatusUI(status, nodeList);
                } else if (MeshWaveService.ACTION_RECEIVE_MESSAGE.equals(action)) {
                    String message = intent.getStringExtra(MeshWaveService.EXTRA_MESSAGE);
                    if (chatFragment != null && chatFragment.isAdded()) {
                        chatFragment.addReceivedMessage(message);
                    }
                }
            }
        };
        IntentFilter filter = new IntentFilter();
        filter.addAction(MeshWaveService.ACTION_UPDATE_UI);
        filter.addAction(MeshWaveService.ACTION_RECEIVE_MESSAGE);
        LocalBroadcastManager.getInstance(this).registerReceiver(uiUpdateReceiver, filter);
    }

    private void checkPrerequisitesAndStart() {
        if (hasPermissions()) {
            initiateBigBang();
        } else {
            requestPermissions();
        }
    }

    private boolean hasPermissions() {
        if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.S) {
            return ActivityCompat.checkSelfPermission(this, Manifest.permission.ACCESS_FINE_LOCATION) == PackageManager.PERMISSION_GRANTED &&

// v0.3.12-alpha
package com.example.meshwave;

import android.annotation.SuppressLint;
import android.app.Notification;
import android.app.NotificationChannel;
import android.app.NotificationManager;
import android.app.Service;
import android.content.Intent;
import android.net.wifi.p2p.WifiP2pConfig;
import android.net.wifi.p2p.WifiP2pDevice;
import android.net.wifi.p2p.WifiP2pInfo;
import android.net.wifi.p2p.WifiP2pManager;
import android.net.wifi.p2p.nsd.WifiP2pDnsSdServiceInfo;
import android.net.wifi.p2p.nsd.WifiP2pDnsSdServiceRequest;
import android.os.Build;
import android.os.Handler;
import android.os.IBinder;
import android.os.Looper;
import android.util.Log;

import androidx.annotation.Nullable;
import androidx.core.app.NotificationCompat;
import androidx.localbroadcastmanager.content.LocalBroadcastManager;

import java.util.HashMap;
import java.util.Map;
import java.util.function.Consumer;

@SuppressLint("MissingPermission")
public class MeshWaveService extends Service {
    private static final String TAG = "MeshWaveService";
    private static final String CHANNEL_ID = "MeshWaveChannel";
    private static final int NOTIFICATION_ID = 1;

    public static final String ACTION_START_SERVICE = "com.example.meshwave.START_SERVICE";
    public static final String ACTION_SEND_MESSAGE = "com.example.meshwave.SEND_MESSAGE";
    public static final String ACTION_UPDATE_UI = "com.example.meshwave.UPDATE_UI";
    public static final String ACTION_RECEIVE_MESSAGE = "com.example.meshwave.RECEIVE_MESSAGE";
    public static final String ACTION_CONNECTION_INFO = "com.example.meshwave.CONNECTION_INFO";

    public static final String EXTRA_SESSID = "sessid";
    public static final String EXTRA_USERNAME = "username";
    public static final String EXTRA_MESSAGE = "message";
    public static final String EXTRA_STATUS_TEXT = "status_text";
    public static final String EXTRA_NODE_LIST = "node_list";

    private WifiP2pManager manager;
    private WifiP2pManager.Channel channel;
    private WifiP2pDnsSdServiceRequest serviceRequest;
    private CommunicationThread communicationThread;
    private String currentSessid;
    private String username;

    @Override
    public void onCreate() {
        super.onCreate();
        manager = (WifiP2pManager) getSystemService(WIFI_P2P_SERVICE);
        channel = manager.initialize(this, getMainLooper(), null);
        createNotificationChannel();
    }

    @Override
    public int onStartCommand(Intent intent, int flags, int startId) {
        if (intent != null) {
            String action = intent.getAction();
            if (ACTION_START_SERVICE.equals(action)) {
                currentSessid = intent.getStringExtra(EXTRA_SESSID);
                username = intent.getStringExtra(EXTRA_USERNAME);
                startForeground(NOTIFICATION_ID, createNotification("Inicializando..."));
                startRegistrationAndDiscovery();
            } else if (ACTION_SEND_MESSAGE.equals(action)) {
                String message = intent.getStringExtra(EXTRA_MESSAGE);
                if (communicationThread != null) {
                    communicationThread.write(message.getBytes());
                }
            } else if (ACTION_CONNECTION_INFO.equals(action)) {
                WifiP2pInfo info;
                if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.TIRAMISU) {
                    info = intent.getParcelableExtra("info", WifiP2pInfo.class);
                } else {
                    info = intent.getParcelableExtra("info");
                }
                if (info != null) {
                    handleConnectionChange(info);
                }
            }
        }
        return START_STICKY;
    }

    private void startRegistrationAndDiscovery() {
        stopServiceDiscovery();

        Map<String, String> record = new HashMap<>();
        record.put("user", username);
        record.put("sessid", currentSessid);

        WifiP2pDnsSdServiceInfo serviceInfo = WifiP2pDnsSdServiceInfo.newInstance(MainActivity.SERVICE_INSTANCE, MainActivity.SERVICE_REG_TYPE, record);
        manager.addLocalService(channel, serviceInfo, new WifiP2pManager.ActionListener() {
            @Override
            public void onSuccess() {
                Log.d(TAG, "Serviço anunciado. Aguardando 1s para iniciar descoberta.");
                new Handler(Looper.getMainLooper()).postDelayed(() -> startServiceDiscoveryOnly(), 1000);
            }
            @Override
            public void onFailure(int reason) {
                Log.e(TAG, "Falha ao anunciar serviço: " + reason);
            }
        });
    }

    private void startServiceDiscoveryOnly() {
        WifiP2pManager.DnsSdTxtRecordListener txtListener = (fullDomain, record, device) -> {
            if (record.containsKey("sessid") && record.get("sessid").equals(currentSessid)) {
                Log.d(TAG, "Nó compatível encontrado: " + device.deviceName);
                connectToDevice(device);
            }
        };

        manager.setDnsSdResponseListeners(channel, null, txtListener);
        serviceRequest = WifiP2pDnsSdServiceRequest.newInstance();
        manager.addServiceRequest(channel, serviceRequest, new WifiP2pManager.ActionListener() {
            @Override
            public void onSuccess() {
                manager.discoverServices(channel, new WifiP2pManager.ActionListener() {
                    @Override
                    public void onSuccess() {
                        Log.d(TAG, "Descoberta de serviços iniciada.");
                        broadcastStateUpdate("Procurando...", "Nenhum nó detectado.");
                    }
                    @Override
                    public void onFailure(int reason) {
                        Log.e(TAG, "Falha ao iniciar descoberta: " + reason);
                    }
                });
            }
            @Override
            public void onFailure(int reason) {
                Log.e(TAG, "Falha ao adicionar requisição de serviço: " + reason);
            }
        });
    }

    private void connectToDevice(WifiP2pDevice device) {
        if (device == null) return;
        WifiP2pConfig config = new WifiP2pConfig();
        config.deviceAddress = device.deviceAddress;
        manager.connect(channel, config, new WifiP2pManager.ActionListener() {
            @Override
            public void onSuccess() { Log.d(TAG, "Enviando pedido de conexão para: " + device.deviceName); }
            @Override
            public void onFailure(int reason) { Log.e(TAG, "Falha ao conectar: " + reason); }
        });
    }

    public void handleConnectionChange(WifiP2pInfo info) {
        if (info.groupFormed) {
            stopServiceDiscovery();
            if (communicationThread != null) {
                communicationThread.close();
            }
            Consumer<String> messageConsumer = this::broadcastReceivedMessage;
            communicationThread = new CommunicationThread(info.groupOwnerAddress, info.isGroupOwner, messageConsumer);
            communicationThread.start();
            String status = info.isGroupOwner ? "Conectado (Líder)" : "Conectado (Cliente)";
            broadcastStateUpdate(status, "Conectado a " + (info.isGroupOwner ? "clientes" : "líder"));
        } else {
            broadcastStateUpdate("Desconectado", "Nenhum nó detectado.");
            startRegistrationAndDiscovery();
        }
    }

    private void stopServiceDiscovery() {
        if (manager != null && channel != null) {
            manager.clearLocalServices(channel, null);
            if (serviceRequest != null) {
                manager.removeServiceRequest(channel, serviceRequest, null);
            }
        }
    }

    private void broadcastReceivedMessage(String message) {
        Intent intent = new Intent(ACTION_RECEIVE_MESSAGE);
        intent.putExtra(EXTRA_MESSAGE, message);

package com.example.meshwave;

import android.annotation.SuppressLint;
import android.app.Notification;
import android.app.NotificationChannel;
import android.app.NotificationManager;
import android.app.Service;
import android.content.Intent;
import android.net.wifi.p2p.WifiP2pConfig;
import android.net.wifi.p2p.WifiP2pDevice;
import android.net.wifi.p2p.WifiP2pInfo;
import android.net.wifi.p2p.WifiP2pManager;
import android.net.wifi.p2p.nsd.WifiP2pDnsSdServiceInfo;
import android.net.wifi.p2p.nsd.WifiP2pDnsSdServiceRequest;
import android.os.Build;
import android.os.Handler;
import android.os.IBinder;
import android.os.Looper;
import android.util.Log;

import androidx.annotation.Nullable;
import androidx.core.app.NotificationCompat;
import androidx.localbroadcastmanager.content.LocalBroadcastManager;

import java.util.HashMap;
import java.util.Map;
import java.util.function.Consumer;

@SuppressLint("MissingPermission")
public class MeshWaveService extends Service {
    private static final String TAG = "MeshWaveService";
    private static final String CHANNEL_ID = "MeshWaveChannel";
    private static final int NOTIFICATION_ID = 1;

    public static final String ACTION_START_SERVICE = "com.example.meshwave.START_SERVICE";
    public static final String ACTION_SEND_MESSAGE = "com.example.meshwave.SEND_MESSAGE";
    public static final String ACTION_UPDATE_UI = "com.example.meshwave.UPDATE_UI";
    public static final String ACTION_RECEIVE_MESSAGE = "com.example.meshwave.RECEIVE_MESSAGE";
    public static final String ACTION_CONNECTION_INFO = "com.example.meshwave.CONNECTION_INFO";

    public static final String EXTRA_SESSID = "sessid";
    public static final String EXTRA_USERNAME = "username";
    public static final String EXTRA_MESSAGE = "message";
    public static final String EXTRA_STATUS_TEXT = "status_text";
    public static final String EXTRA_NODE_LIST = "node_list";

    private WifiP2pManager manager;
    private WifiP2pManager.Channel channel;
    private WifiP2pDnsSdServiceRequest serviceRequest;
    private CommunicationThread communicationThread;
    private String currentSessid;
    private String username;

    @Override
    public void onCreate() {
        super.onCreate();
        manager = (WifiP2pManager) getSystemService(WIFI_P2P_SERVICE);
        channel = manager.initialize(this, getMainLooper(), null);
        createNotificationChannel();
    }

    @Override
    public int onStartCommand(Intent intent, int flags, int startId) {
        if (intent != null) {
            String action = intent.getAction();
            if (ACTION_START_SERVICE.equals(action)) {
                currentSessid = intent.getStringExtra(EXTRA_SESSID);
                username = intent.getStringExtra(EXTRA_USERNAME);
                startForeground(NOTIFICATION_ID, createNotification("Inicializando..."));
                startRegistrationAndDiscovery();
            } else if (ACTION_SEND_MESSAGE.equals(action)) {
                String message = intent.getStringExtra(EXTRA_MESSAGE);
                if (communicationThread != null) {
                    communicationThread.write(message.getBytes());
                }
            } else if (ACTION_CONNECTION_INFO.equals(action)) {
                WifiP2pInfo info;
                if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.TIRAMISU) {
                    info = intent.getParcelableExtra("info", WifiP2pInfo.class);
                } else {
                    info = intent.getParcelableExtra("info");
                }
                if (info != null) {
                    handleConnectionChange(info);
                }
            }
        }
        return START_STICKY;
    }

    private void startRegistrationAndDiscovery() {
        stopServiceDiscovery();

        Map<String, String> record = new HashMap<>();
        record.put("user", username);
        record.put("sessid", currentSessid);

        WifiP2pDnsSdServiceInfo serviceInfo = WifiP2pDnsSdServiceInfo.newInstance(MainActivity.SERVICE_INSTANCE, MainActivity.SERVICE_REG_TYPE, record);
        manager.addLocalService(channel, serviceInfo, new WifiP2pManager.ActionListener() {
            @Override
            public void onSuccess() {
                Log.d(TAG, "Serviço anunciado. Aguardando 1s para iniciar descoberta.");
                new Handler(Looper.getMainLooper()).postDelayed(() -> startServiceDiscoveryOnly(), 1000);
            }
            @Override
            public void onFailure(int reason) {
                Log.e(TAG, "Falha ao anunciar serviço: " + reason);
            }
        });
    }

    private void startServiceDiscoveryOnly() {
        WifiP2pManager.DnsSdTxtRecordListener txtListener = (fullDomain, record, device) -> {
            if (record.containsKey("sessid") && record.get("sessid").equals(currentSessid)) {
                Log.d(TAG, "Nó compatível encontrado: " + device.deviceName);
                connectToDevice(device);
            }
        };

        manager.setDnsSdResponseListeners(channel, null, txtListener);
        serviceRequest = WifiP2pDnsSdServiceRequest.newInstance();
        manager.addServiceRequest(channel, serviceRequest, new WifiP2pManager.ActionListener() {
            @Override
            public void onSuccess() {
                manager.discoverServices(channel, new WifiP2pManager.ActionListener() {
                    @Override
                    public void onSuccess() {
                        Log.d(TAG, "Descoberta de serviços iniciada.");
                        broadcastStateUpdate("Procurando...", "Nenhum nó detectado.");
                    }
                    @Override
                    public void onFailure(int reason) {
                        Log.e(TAG, "Falha ao iniciar descoberta: " + reason);
                    }
                });
            }
            @Override
            public void onFailure(int reason) {
                Log.e(TAG, "Falha ao adicionar requisição de serviço: " + reason);
            }
        });
    }

    private void connectToDevice(WifiP2pDevice device) {
        if (device == null) return;
        WifiP2pConfig config = new WifiP2pConfig();
        config.deviceAddress = device.deviceAddress;
        manager.connect(channel, config, new WifiP2pManager.ActionListener() {
            @Override
            public void onSuccess() { Log.d(TAG, "Enviando pedido de conexão para: " + device.deviceName); }
            @Override
            public void onFailure(int reason) { Log.e(TAG, "Falha ao conectar: " + reason); }
        });
    }

    public void handleConnectionChange(WifiP2pInfo info) {
        if (info.groupFormed) {
            stopServiceDiscovery();
            if (communicationThread != null) {
                communicationThread.close();
            }
            Consumer<String> messageConsumer = this::broadcastReceivedMessage;
            communicationThread = new CommunicationThread(info.groupOwnerAddress, info.isGroupOwner, messageConsumer);
            communicationThread.start();
            String status = info.isGroupOwner ? "Conectado (Líder)" : "Conectado (Cliente)";
            broadcastStateUpdate(status, "Conectado a " + (info.isGroupOwner ? "clientes" : "líder"));
        } else {
            broadcastStateUpdate("Desconectado", "Nenhum nó detectado.");
            startRegistrationAndDiscovery();
        }
    }

    private void stopServiceDiscovery() {
        if (manager != null && channel != null) {
            manager.clearLocalServices(channel, null);
            if (serviceRequest != null) {
                manager.removeServiceRequest(channel, serviceRequest, null);
            }
        }
    }

    private void broadcastReceivedMessage(String message) {
        Intent intent = new Intent(ACTION_RECEIVE_MESSAGE);
        intent.putExtra(EXTRA_MESSAGE, message);
        LocalBroadcastManager.getInstance(this).sendBroadcast(intent);
    }

    private void broadcastStateUpdate(String status, String nodeList) {
        Intent intent = new Intent(ACTION_UPDATE_UI);
        intent.putExtra(EXTRA_STATUS_TEXT, status);
        intent.putExtra(EXTRA_NODE_LIST, nodeList);
        LocalBroadcastManager.getInstance(this).sendBroadcast(intent);
    }

    private void createNotificationChannel() {
        if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.O) {
            NotificationChannel serviceChannel = new NotificationChannel(
                    CHANNEL_ID, "MeshWave Service Channel", NotificationManager.IMPORTANCE_DEFAULT);
            getSystemService(NotificationManager.class).createNotificationChannel(serviceChannel);
        }
    }

    private Notification createNotification(String text) {
        return new NotificationCompat.Builder(this, CHANNEL_ID)
                .setContentTitle("MeshWave Ativo")
                .setContentText(text)
                .setSmallIcon(R.drawable.ic_status)
                .build();
    }

    @Override
    public void onDestroy() {
        super.onDestroy();
        if (communicationThread != null) {
            communicationThread.close();
        }
        stopServiceDiscovery();
        if (manager != null && channel != null) {
            manager.removeGroup(channel, null);
        }
    }

    @Nullable
    @Override
    public IBinder onBind(Intent intent) {
        return null;
    }
}
