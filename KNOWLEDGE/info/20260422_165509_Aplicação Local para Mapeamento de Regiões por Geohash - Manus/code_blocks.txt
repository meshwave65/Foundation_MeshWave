# app.py (VERSÃO FINAL COM ENTRADA POR GEOHASH )

from flask import Flask, render_template, request, jsonify, send_file
import pandas as pd
import io
import geo_logic
import pygeohash

app = Flask(__name__)

@app.route('/')
def index():
    return render_template('index.html')

# ROTA PARA O MODO AUTOMÁTICO (EXPLORAÇÃO)
@app.route('/api/generate_auto_grid', methods=['POST'])
def generate_auto_grid_api():
    data = request.json
    cidade = data.get('cidade')
    pais = data.get('pais')
    if not cidade or not pais:
        return jsonify({'error': 'Cidade e país são obrigatórios.'}), 400

    dados_cidade = geo_logic.buscar_dados_cidade(cidade, pais)
    if not dados_cidade:
        return jsonify({'error': f'Não foi possível encontrar dados detalhados para "{cidade}".'}), 400

    ponto_origem = dados_cidade['ponto_central']
    geohash_origem = pygeohash.encode(ponto_origem['lat'], ponto_origem['lon'], precision=geo_logic.PRECISAO_GEOHASH)
    lista_geohashes = geo_logic.gerar_geohashes_por_poligono(dados_cidade['limites'], dados_cidade['poligono'])
    
    return processar_e_retornar_grade(lista_geohashes, geohash_origem)

# ROTA PARA O MODO MANUAL (POR GEOHASH)
@app.route('/api/generate_manual_grid', methods=['POST'])
def generate_manual_grid_api():
    data = request.json
    try:
        geohash_origem = data['geohash_origem'].strip().lower()
        raio_x, raio_y = int(data['raio_x']), int(data['raio_y'])
        
        # Decodifica o geohash para obter a lat/lon central
        lat, lon = pygeohash.decode(geohash_origem)

    except (ValueError, TypeError, KeyError):
        return jsonify({'error': 'Dados inválidos. Verifique o Geohash e os raios inseridos.'}), 400

    lista_geohashes = geo_logic.gerar_geohashes_por_raio(lat, lon, raio_x, raio_y)

    return processar_e_retornar_grade(lista_geohashes, geohash_origem)

def processar_e_retornar_grade(lista_geohashes, geohash_origem):
    """Função auxiliar comum para processar a lista de geohashes e retornar o JSON."""
    dados_grade = []
    for gh in lista_geohashes:
        coords = geo_logic.converter_para_grade_customizada(gh, geohash_origem)
        if coords[0] is None: continue
        try:
            bbox = pygeohash.decode_exactly(gh)
            bounds_leaflet = [[bbox[0] - bbox[2], bbox[1] - bbox[3]], [bbox[0] + bbox[2], bbox[1] + bbox[3]]]
            dados_grade.append({
                'geohash': gh, 'coord_x': coords[0], 'coord_y': coords[1], 'bounds': bounds_leaflet
            })
        except Exception:
            continue
    
    app.config['LAST_GRID_DATA'] = dados_grade
    return jsonify({'geohash_origem': geohash_origem, 'dados_grade': dados_grade})

@app.route('/api/export_excel')
def export_excel():
    dados_grade = app.config.get('LAST_GRID_DATA')
    if not dados_grade:
        return "Nenhum dado de grade para exportar.", 404
    df = pd.DataFrame(dados_grade)
    df = df[['geohash', 'coord_x', 'coord_y']]
    output = io.BytesIO()
    with pd.ExcelWriter(output, engine='xlsxwriter') as writer:
        df.to_excel(writer, index=False, sheet_name='Grade')
    output.seek(0)
    return send_file(
        output,
        mimetype='application/vnd.openxmlformats-officedocument.spreadsheetml.sheet',
        as_attachment=True,
        download_name='grade_geografica.xlsx'
    )

if __name__ == '__main__':
    app.run(debug=True)

// static/js/main.js (VERSÃO FINAL COM ENTRADA POR GEOHASH)

function openTab(evt, tabName) {
    let i, tabcontent, tablinks;
    tabcontent = document.getElementsByClassName("tab-content");
    for (i = 0; i < tabcontent.length; i++) {
        tabcontent[i].style.display = "none";
    }
    tablinks = document.getElementsByClassName("tab-button");
    for (i = 0; i < tablinks.length; i++) {
        tablinks[i].className = tablinks[i].className.replace(" active", "");
    }
    document.getElementById(tabName).style.display = "block";
    evt.currentTarget.className += " active";
}

document.addEventListener('DOMContentLoaded', function() {
    const exportBtn = document.getElementById('export-btn');
    const map = L.map('map').setView([-22.906, -43.172], 13);
    let featureGroup = L.featureGroup().addTo(map);
    
    L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', {
        attribution: '&copy; <a href="https://www.openstreetmap.org/copyright">OpenStreetMap</a> contributors'
    } ).addTo(map);

    // Modo Automático
    const generateAutoBtn = document.getElementById('generate-auto-btn');
    const cidadeInput = document.getElementById('cidade');
    const paisInput = document.getElementById('pais');

    // Modo Manual
    const generateManualBtn = document.getElementById('generate-manual-btn');
    const geohashOrigemInput = document.getElementById('geohash_origem');
    const raioXInput = document.getElementById('raio_x');
    const raioYInput = document.getElementById('raio_y');

    // Listeners dos Botões
    generateAutoBtn.addEventListener('click', () => {
        const payload = { cidade: cidadeInput.value, pais: paisInput.value };
        gerarGrade('/api/generate_auto_grid', payload);
    });

    generateManualBtn.addEventListener('click', () => {
        const payload = {
            geohash_origem: geohashOrigemInput.value,
            raio_x: raioXInput.value,
            raio_y: raioYInput.value
        };
        gerarGrade('/api/generate_manual_grid', payload);
    });

    // Função Genérica para Gerar a Grade
    async function gerarGrade(apiUrl, payload) {
        featureGroup.clearLayers();
        exportBtn.style.display = 'none';
        alert('Gerando a grade... Por favor, aguarde.');

        try {
            const response = await fetch(apiUrl, {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify(payload)
            });

            const data = await response.json();
            if (!response.ok) throw new Error(data.error || 'Erro desconhecido');

            data.dados_grade.forEach(cell => {
                const isOrigem = cell.geohash === data.geohash_origem;
                const color = isOrigem ? "#FF0000" : "#ff7800";
                const rectangle = L.rectangle(cell.bounds, {
                    color: color,
                    weight: isOrigem ? 2 : 1,
                    fillOpacity: 0.2
                });
                
                // Adiciona um evento de clique para copiar o geohash
                rectangle.on('click', () => {
                    navigator.clipboard.writeText(cell.geohash).then(() => {
                        alert(`Geohash "${cell.geohash}" copiado para a área de transferência!`);
                    });
                });

                const tooltipContent = `<b>CLA: (${cell.coord_x}, ${cell.coord_y})</b>  
Geohash: ${cell.geohash}  
<i>(clique para copiar)</i>`;
                rectangle.bindTooltip(tooltipContent).addTo(featureGroup);
            });
            
            if (data.dados_grade.length > 0) {
                map.fitBounds(featureGroup.getBounds().pad(0.1));
                exportBtn.style.display = 'inline-block';
            } else {
                alert("Nenhuma célula foi gerada para os parâmetros informados.");
            }

        } catch (error) {
            alert(`Erro ao gerar a grade: ${error.message}`);
        }
    }
});
