export async function createMceLead(data: { nome: string; telefone: string; email: string }) {
  console.log("Tentando gravar lead com dados:", data); // Ver o que está saindo
  
  const { data: inserted, error } = await supabase
    .from('leads')
    .insert({
      nome: data.nome,
      telefone: data.telefone,
      email: data.email,
    })
    .select(); // Pega o objeto inteiro de volta

  if (error) {
    console.error("ERRO REAL DO SUPABASE NO LEAD:", error); // Isso vai brilhar no seu terminal

}
